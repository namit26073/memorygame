const suits = ["spades", "clubs", "hearts", "diamonds"];
const nums = [
  "ace",
  "2",
  "3",
  "4",
  "5",
  "6",
  "7",
  "8",
  "9",
  "10",
  "jack",
  "queen",
  "king",
];
let displayedCards = new Array(8).fill(0);

/**
 * creating a class for our deck of cards
 *
 *
 *
 *
 *
 */
export class Deck {
  constructor(cards = allCards()) {
    this.cards = cards; // assigns all cards to attribute of card
  }

  //returns number of cards in deck
  get numberofcards() {
    return this.cards.length;
  }

  //shuffles deck so each game cards are different
  shuffle() {
    this.cards.sort(() => Math.random() - 0.5);
  }
}

// creating a class from all the cards we need
class card {
  constructor(suit, number) {
    this.suit = suit; //the attributes needed for cards
    this.num = number;
  }
}

// creating a function which creates a full deck of cards
function allCards() {
  return suits.flatMap((suit) => {
    return nums.map((num) => {
      return new card(suit, num); //for each suit, it assigns a number so we get all the cards
    });
  });
}

/**
 *takes input array and uses random function to swap card with another random card in the deck
 * @param {Array} orderedArray array of all the cards
 * @returns {Array} the randomized array
 */

export function randomize(orderedArray) {
  orderedArray.sort(() => Math.random() - 0.5);
  return orderedArray;
}

export default function createSet() {
  let tempDeck = new Deck();
  tempDeck.shuffle();
  displayedCards.forEach((currentcard, pointer) => {
    displayedCards[pointer] = tempDeck.cards[pointer];
  });
  let doubledCards = displayedCards.concat(displayedCards);
  const randomArray = randomize(doubledCards);
  return randomArray;
}
